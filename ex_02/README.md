# 행렬의 부분구간의 종료값 -1이라는 의미

---

	>>> import numpy as np
	>>> A = np.array([[1.0, 2.0, 7.0], [2.0, 1.0, 0.0], [4.0, 2.0, 3.0]])
	>>> A
	array([[1., 2., 7.],
	       [2., 1., 0.],
	       [4., 2., 3.]])
	>>> A[1, :]
	array([2., 1., 0.])
	>>> A[:, 2]
	array([7., 0., 3.])
	>>> A[1:2, 1:2]
	array([[1.]])
	>>> A[1:3, 1:3]
	array([[1., 0.],
	       [2., 3.]])
A[1:2, 1:2] 는 1행 1열임. 2에서 -1

# np.copy()를 해야 값이 복사가 됨. 

---

>>> A
array([[1., 2., 7.],
       [2., 1., 0.],
       [4., 2., 3.]])
>>> C = np.copy(A)
>>> C
array([[1., 2., 7.],
       [2., 1., 0.],
       [4., 2., 3.]])
>>> B = A
>>> B
array([[1., 2., 7.],
       [2., 1., 0.],
       [4., 2., 3.]])
>>> A[0, :] = np.zeros((1,3))
>>> 
>>> A
array([[0., 0., 0.],
       [2., 1., 0.],
       [4., 2., 3.]])
>>> B
array([[0., 0., 0.],
       [2., 1., 0.],
       [4., 2., 3.]])
